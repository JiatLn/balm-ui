### Default Usage

```scss
@use 'balm-ui/dist/balm-ui';
```

```js
import Vue from 'vue';
import BalmUI from 'balm-ui/dist/balm-ui';

Vue.use(BalmUI);
```

### Individual Usage

```scss
@use 'balm-ui/components/core';
@use 'balm-ui/components/divider/divider';
```

```js
import Vue from 'vue';
import UiDivider from 'balm-ui/components/divider';

Vue.use(UiDivider);
```
