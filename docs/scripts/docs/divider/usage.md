### Default Usage

```css
@use 'balm-ui/dist/balm-ui.scss';
```

```js
import Vue from 'vue';
import BalmUI from 'balm-ui/dist/balm-ui';

Vue.use(BalmUI);
```

### Individual Usage

```css
@use 'balm-ui/components/core.scss';
@use 'balm-ui/components/divider/divider.scss';
```

```js
import Vue from 'vue';
import UiDivider from 'balm-ui/components/divider';

Vue.use(UiDivider);
```
